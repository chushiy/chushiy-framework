ALTER TABLE 表名
ADD COLUMN `delete_flag` INT NOT NULL DEFAULT 0 COMMENT '删除标志（0代表未删除，1代表已删除）',
ADD COLUMN `create_by` VARCHAR(50) NOT NULL COMMENT '创建人',
ADD COLUMN `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '创建时间',
ADD COLUMN `update_by` VARCHAR(50) NOT NULL COMMENT '更新人',
ADD COLUMN `update_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP COMMENT '更新时间',
